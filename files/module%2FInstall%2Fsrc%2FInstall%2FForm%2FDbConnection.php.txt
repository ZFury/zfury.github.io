namespace Install\Form;

use Zend\Form\Form;

class DbConnection extends Form
{
    public function __construct()
    {
        parent::__construct('db_connection');

        $this->add([
            'name' => 'host',
            'type' => 'Text',
            'options' => [
                'label' => 'Host',
            ],
        ]);

        $this->add([
            'name' => 'port',
            'type' => 'Number',
            'options' => [
                'label' => 'Port',
            ],
        ]);

        $this->add([
            'name' => 'user',
            'type' => 'Text',
            'options' => [
                'label' => 'User',
            ],
        ]);

        $this->add([
            'name' => 'password',
            'type' => 'Text',
            'options' => [
                'label' => 'Password',
            ],
        ]);

        $this->add([
            'name' => 'dbname',
            'type' => 'Text',
            'options' => [
                'label' => 'DB name',
            ],
        ]);
    }
}
